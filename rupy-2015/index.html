mrbgems

mruby-cool-ext/src/extension.c


#include "mruby/variable.h"

static mrb_value
mrb_bomb_plant(mrb_state *mrb, mrb_value self)
{
  mrb_int time_value;
  mrb_get_args(mrb, "i", &time_value);
  mrb_sym time_sym = mrb_intern_cstr(mrb, "@tempo");
  mrb_iv_set(mrb, self, time_sym, mrb_fixnum_value(time_value));
  return mrb_nil_value();
}
